    class Solution {
        int count = 0;
        static final char LAND = '1';
        static final char WATER = '0';
        public int numIslands(char[][] grid) {
            int m = grid.length;
            int n = grid[0].length;

            for(int i=0; i<m; i++) {
                for(int j=0; j<n; j++) {
                    if(grid[i][j]==LAND) {
                        removeIsland(grid,i,j);
                        count++;
                    }
                }
            }
            return count;
        }

        public void removeIsland(char[][] grid, int i, int j) {

            if (i>=grid.length || j>=grid[0].length || i<0 || j<0 || grid[i][j]==WATER) {
                return;
            }

            if (grid[i][j] == LAND) {
                grid[i][j] = WATER;
                removeIsland(grid, i+1, j);
                removeIsland(grid, i, j+1);
                removeIsland(grid, i-1, j);
                removeIsland(grid, i, j-1);
            }
        }
    }
